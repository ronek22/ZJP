def is_pangram(sentence):
    sentence = sentence.lower()
    if all(ord(char) < 128 for char in sentence):
        count = {}
        total = 0
        for char in sentence:
            if 'a'<=char<='z' and char not in count:
                count[char] = 1
                total +=ord(char)
        if total == 2847:
            return True
        else:
            return False
            
    else:
        raise Exception('sentence format error')
    pass
